       const mobileMenuButton = document.getElementById('mobile-menu-button');
       const mobileMenu = document.getElementById('mobile-menu');
       mobileMenuButton.addEventListener('click', () => {
           mobileMenu.classList.toggle('mobile-menu-hidden');
       });

       const pageLinks = document.querySelectorAll('.page-link');
       const navItems = document.querySelectorAll('.nav-item');
       const pageContents = document.querySelectorAll('.page-content');

       function showPage(targetId) {
           pageContents.forEach(content => {
               if (content.id === targetId) {
                   content.classList.remove('hidden');
                   const navHeight = document.querySelector('nav').offsetHeight;
                   const elementPosition = content.getBoundingClientRect().top + window.pageYOffset;
                   const offsetPosition = elementPosition - navHeight - 20;
                   window.scrollTo({ top: offsetPosition, behavior: "smooth" });
               } else {
                   content.classList.add('hidden');
               }
           });

           navItems.forEach(item => {
               item.classList.remove('nav-active', 'text-blue-600', 'font-semibold');
               if (item.getAttribute('href') === `#${targetId}`) {
                   item.classList.add('nav-active', 'text-blue-600', 'font-semibold');
               } else {
                    item.classList.remove('text-blue-600', 'font-semibold');
               }
           });

           if (!mobileMenu.classList.contains('mobile-menu-hidden')) {
               mobileMenu.classList.add('mobile-menu-hidden');
           }
       }

       pageLinks.forEach(link => {
           link.addEventListener('click', (event) => {
               event.preventDefault();
               const targetId = link.getAttribute('href').substring(1);
               if (document.getElementById(targetId)) {
                   showPage(targetId);
                   history.pushState(null, null, `#${targetId}`);
               } else {
                   console.warn('Section not found:', targetId);
               }
           });
       });
       
       function loadInitialPage() {
           let targetId = window.location.hash.substring(1);
           const validSections = Array.from(pageContents).map(pc => pc.id);
           if (!targetId || !validSections.includes(targetId)) {
               targetId = 'accueil';
           }
           showPage(targetId);
       }

       window.addEventListener('load', loadInitialPage);
       window.addEventListener('popstate', loadInitialPage);

       document.getElementById('currentYear').textContent = new Date().getFullYear();

       const confirmationMessage = document.getElementById('confirmationMessage');
       if (contactForm) {
           contactForm.addEventListener('submit', function(event) {
               event.preventDefault();
               console.log('Formulaire soumis (simulation)');
               confirmationMessage.textContent = "Votre message m'a bien été envoyé ! Je vous répondrai bientôt.";
               confirmationMessage.style.display = 'block';
               contactForm.reset();
               setTimeout(() => {
                   confirmationMessage.style.display = 'none';
               }, 5000);
           });
       }
